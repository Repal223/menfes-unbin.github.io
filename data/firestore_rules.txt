rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Default deny
    match /{document=**} {
      allow read, write: if false;
    }

    // Allow client to read own notifications only (if you enable Firebase Auth)
    match /notifications/{id} {
      allow read: if request.auth != null && resource.data.receiver_uid == request.auth.uid;
      allow update: if request.auth != null && request.resource.data.read == true && resource.data.receiver_uid == request.auth.uid;
      allow create, delete: if false;
    }
  }
}
